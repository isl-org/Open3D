# This is a config file for the ICP Odometry Examples.
# When voxel_size is -1, no downsampling is performed.
# To run DemoTICPOdom:
# 1. Run script download_kitti_samples.sh.
# 2. Go to build/bin/examples
# 3. Run the following command:
# ./DemoTICPOdom CPU:0 ../../../examples/cpp/registration_example_util/TICPOdomConfig.txt
# Change CPU:0 to CUDA:0 for running it on primary GPU.

# option to turn ON / OFF visualization:
visualization = ON

# Verbosity can be Info, Debug.
verbosity = Info

# Path of the downloaded dataset containing frames in .pcd format, named
# sequentially from 0.pcd to [end_range - 1].pcd.
 dataset_path = ../../open3d_downloads/kitti_samples/
# dataset_path = /home/rey/Datasets/CARLA/pcd/


# Range of frames is 0 to end_range - 1.
end_range = 400

# Registration method can be PointToPoint or PointToPlane.
registration_method = PointToPlane

# Multi-Scale ICP parameters:

# Scale 1:
 voxel_size = 0.8
 search_radii = 1.2
 criteria.relative_fitness = 0.01
 criteria.relative_rmse = 0.01
 criteria.max_iterations = 10

# Scale 2:
 voxel_size = 0.5
 search_radii = 1.0
 criteria.relative_fitness = 0.001
 criteria.relative_rmse = 0.001
 criteria.max_iterations = 10
